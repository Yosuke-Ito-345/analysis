    # Normal distribution ``` normal_peak s := (sqrtr (s ^+ 2 * pi *+ 2))^-1 normal_fun m s x := expR (- (x - m) ^+ 2 / (s ^+ 2 *+ 2)) normal_pdf m s == pdf of the normal distribution with mean m and standard deviation s Using normal_peak and normal_pdf. normal_prob m s == normal probability measure ```

    Definition normal_fun m s x := expR (- (x - m) ^+ 2 / (s ^+ 2 *+ 2)).

    Lemma measurable_normal_fun m s : measurable_fun [set :R] (normal_fun m s).

    Definition normal_prob {R : realType} (m : R) (s : R) : set _ -> \bar R :=
      fun V => (\int[lebesgue_measure]_(x in V) (normal_pdf m s x)%:E)%E.
